# Import necessary libraries
import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

# Define a main function to encapsulate the code
def main():
    # Create a simple dataset for demonstration
    # Generate x values
    x = np.arange(1, 11)
    
    # Generate y values (squares of x)
    y = x ** 2
    
    # Create a pandas DataFrame for better data management
    df = pd.DataFrame({
        'X': x,
        'Y': y
    })
    
    # Print the DataFrame for visibility
    print("Dataset:")
    print(df)
    
    # Create a line chart
    plt.figure(figsize=(10, 6))  # Set chart size
    plt.plot(df['X'], df['Y'], marker='o')  # Plot with markers
    plt.title('Line Chart: Y = X^2')  # Set chart title
    plt.xlabel('X')  # Set x-axis label
    plt.ylabel('Y')  # Set y-axis label
    plt.grid(True)  # Show grid for better readability
    plt.show()  # Display the chart
    
    # Create a bar chart
    plt.figure(figsize=(10, 6))  # Set chart size
    plt.bar(df['X'], df['Y'], color='skyblue')  # Plot bars
    plt.title('Bar Chart: Y = X^2')  # Set chart title
    plt.xlabel('X')  # Set x-axis label
    plt.ylabel('Y')  # Set y-axis label
    plt.show()  # Display the chart
    
    # Create a scatter plot
    plt.figure(figsize=(10, 6))  # Set chart size
    plt.scatter(df['X'], df['Y'], color='red')  # Plot points
    plt.title('Scatter Plot: Y = X^2')  # Set chart title
    plt.xlabel('X')  # Set x-axis label
    plt.ylabel('Y')  # Set y-axis label
    plt.show()  # Display the chart

# Call the main function to execute the code
if __name__ == "__main__":
    main()
