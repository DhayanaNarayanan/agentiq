# Import necessary libraries
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# Function to generate sample sales data
def generate_sales_data():
    # Initialize lists to store data
    dates = []
    revenues = []

    # Generate sample data for 12 months
    for i in range(12):
        # Calculate date for current month
        date = datetime(2022, i + 1, 1)

        # Generate random revenue for current month
        revenue = np.random.randint(1000, 10000)

        # Append data to lists
        dates.append(date)
        revenues.append(revenue)

    # Create a pandas DataFrame from the data
    sales_data = pd.DataFrame({
        'Date': dates,
        'Revenue': revenues
    })

    return sales_data

# Function to calculate total revenue by month
def calculate_total_revenue(sales_data):
    # Group sales data by month and calculate total revenue
    total_revenue_by_month = sales_data.groupby(pd.Grouper(key='Date', freq='M'))['Revenue'].sum().reset_index()

    return total_revenue_by_month

# Main function
def main():
    # Generate sample sales data
    sales_data = generate_sales_data()

    # Calculate total revenue by month
    total_revenue_by_month = calculate_total_revenue(sales_data)

    # Print the results
    print("Sample Sales Data:")
    print(sales_data)
    print("\nTotal Revenue by Month:")
    print(total_revenue_by_month)

# Call the main function
if __name__ == "__main__":
    main()
